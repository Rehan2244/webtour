Quick Start

Get started with just a few lines of code:

// Initialize the tour
const tour = new GuidedTour({
  steps: [
    {
      element: '#header',
      title: 'Welcome!',
      content: 'This is your application header',
      position: 'bottom'
    },
    {
      element: '.search-box',
      title: 'Search Feature',
      content: 'Use this to search for anything',
      position: 'top'
    }
  ]
});

// Start the tour
tour.start();
